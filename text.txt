<body>

    <div id="main-text">Six</div>

    <div class="buttons">
        <button id="yes-btn" onclick="handleYes()">Yes</button>
        <button id="no-btn" onmouseover="moveNoButton()" onclick="moveNoButton()">No</button>
    </div>

    <script>
        let currentStep = 0;
        const steps = ["Seven", "Eight", "Nine", "Will you be my valentine?"];
        const mainText = document.getElementById('main-text');
        const yesBtn = document.getElementById('yes-btn');
        const noBtn = document.getElementById('no-btn');

        function handleYes() {
            if (currentStep < steps.length) {
                mainText.innerText = steps[currentStep];
                currentStep++;
                
                if (currentStep === steps.length) {
                    yesBtn.style.transform = "scale(1.4)";
                }
            } else {
                mainText.innerText = "I love you so much! ❤️";
                yesBtn.style.display = "none";
                noBtn.style.display = "none";
                startHearts();
            }
        }

        function moveNoButton() {
            const padding = 50;
            const x = Math.random() * (window.innerWidth - noBtn.offsetWidth - padding);
            const y = Math.random() * (window.innerHeight - noBtn.offsetHeight - padding);
            
            noBtn.style.position = 'fixed';
            noBtn.style.left = x + 'px';
            noBtn.style.top = y + 'px';
        }

        function startHearts() {
            setInterval(() => {
                const heart = document.createElement('div');
                heart.innerHTML = '❤️';
                heart.classList.add('heart');
                heart.style.left = Math.random() * 100 + 'vw';
                heart.style.animationDuration = (Math.random() * 3 + 2) + 's';
                heart.style.opacity = Math.random();
                heart.style.fontSize = (Math.random() * 20 + 10) + 'px';
                document.body.appendChild(heart);
                
                setTimeout(() => { heart.remove(); }, 5000);
            }, 150);
        }
    </script>
